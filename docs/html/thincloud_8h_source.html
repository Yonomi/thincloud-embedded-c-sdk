295 {
296  if (payload == NULL || payloadLen == 0)

298  FUNC_EXIT_RC(SUCCESS);

300

303  json_object *obj = json_tokener_parse_ex(tok, payload, payloadLen);
304  if (obj == NULL)

307  }
308 
309  if (requestId != NULL)

316  }
317 
318  json_object *result = json_object_object_get(obj, "result");
319 
320  if (result != NULL)
321  {
322  if (statusCode != NULL)
323  {
324  *statusCode = json_object_get_int(json_object_object_get(result, "statusCode"));
325  }
326 
327  if (deviceId != NULL)
328  {
329  const char *deviceIdValue = json_object_get_string(json_object_object_get(result, "deviceId"));
330  if (deviceIdValue != NULL)
331  {
332  strcpy(deviceId, deviceIdValue);
333  }
334  }
335  }
336 
337  json_tokener_free(tok);
338  json_object_put(obj);
339 
340  FUNC_EXIT_RC(SUCCESS);
341 }

582 IoT_Error_t send_command_response(AWS_IoT_Client *client, const char *deviceId, const char *commandId, uint16_t statusCode, bool isErrorResponse, char *errorMessage, json_object *body)
583 {
584  char topic[MAX_TOPIC_LENGTH];
585 
586  int topicLen = command_response_topic(topic, deviceId, commandId);
587  if (topicLen <= 0)
588  {
589  FUNC_EXIT_RC(INVALID_TOPIC_TYPE_ERROR);
590  }
591 
592  char payload[MAX_JSON_TOKEN_EXPECTED];
593 
594  IoT_Error_t rc = command_response(payload, commandId, statusCode, isErrorResponse, errorMessage, body);
595 
596  if (rc != SUCCESS)
597  {
598  FUNC_EXIT_RC(rc);
599  }
600 
601  IoT_Publish_Message_Params params;
602  params.qos = QOS0;
603  params.isRetained = false;
604  params.payload = (void *)payload;
605  params.payloadLen = strlen(payload);
606 
607  return aws_iot_mqtt_publish(client, topic, topicLen, &params);
608 }
