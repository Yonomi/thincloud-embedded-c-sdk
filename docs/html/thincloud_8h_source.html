282 IoT_Error_t commissioning_response(char *deviceId, uint16_t *statusCode, char *requestId, char *payload, uint16_t payloadLen)
283 {
284  if (payload == NULL || payloadLen == 0)
285  {
286  FUNC_EXIT_RC(SUCCESS);
287  }
288 
289  json_tokener *tok = json_tokener_new();
290 
291  json_object *obj = json_tokener_parse_ex(tok, payload, payloadLen);
292  if (obj == NULL)
293  {
294  FUNC_EXIT_RC(JSON_PARSE_ERROR);
295  }
296 
297  if (requestId != NULL)
298  {
299  const char *requestIdValue = json_object_get_string(json_object_object_get(obj, "id"));
300  if (requestIdValue != NULL)
301  {
302  strcpy(requestId, requestIdValue);
303  }
304  }
305 
306  json_object *result = json_object_object_get(obj, "result");
307 
308  if (result != NULL)
309  {
310  if (statusCode != NULL)
311  {
312  *statusCode = json_object_get_int(json_object_object_get(result, "statusCode"));
313  }
314 
315  if (deviceId != NULL)
316  {
317  const char *deviceIdValue = json_object_get_string(json_object_object_get(result, "deviceId"));
318  if (deviceIdValue != NULL)
319  {
320  strcpy(deviceId, deviceIdValue);
321  }
322  }
323  }
324 
325  json_tokener_free(tok);
326  json_object_put(obj);
327 
328  FUNC_EXIT_RC(SUCCESS);
329 }

570 IoT_Error_t send_command_response(AWS_IoT_Client *client, const char *deviceId, const char *commandId, uint16_t statusCode, bool isErrorResponse, char *errorMessage, json_object *body)
571 {
572  char topic[MAX_TOPIC_LENGTH];
573 
574  int topicLen = command_response_topic(topic, deviceId, commandId);
575  if (topicLen <= 0)
576  {
577  FUNC_EXIT_RC(INVALID_TOPIC_TYPE_ERROR);
578  }
579 
580  char payload[MAX_JSON_TOKEN_EXPECTED];
581 
582  IoT_Error_t rc = command_response(payload, commandId, statusCode, isErrorResponse, errorMessage, body);
583 
584  if (rc != SUCCESS)
585  {
586  FUNC_EXIT_RC(rc);
587  }
588 
589  IoT_Publish_Message_Params params;
590  params.qos = QOS0;
591  params.isRetained = false;
592  params.payload = (void *)payload;
593  params.payloadLen = strlen(payload);
594 
595  return aws_iot_mqtt_publish(client, topic, topicLen, &params);
596 }
